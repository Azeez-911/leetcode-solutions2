// LeetCode #27 - Remove Element
// Difficulty : Easy
// Language   : unknown
// Runtime    : 0 ms
// Memory     : 10.40 MB
// URL        : https://leetcode.com/problems/remove-element/

int removeElement(int* nums, int numsSize, int val) {
    
    int k = 0;

    for(int i = 0; i < numsSize; i++) {
        
        if(nums[i] != val) {
            nums[k] = nums[i];
            k++;
        }
    }

    return k;
}
